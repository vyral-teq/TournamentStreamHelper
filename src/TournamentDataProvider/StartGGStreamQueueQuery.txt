query StreamQueueQuery($slug: String) {
    event(slug: $slug){
        tournament {
            streamQueue {
                sets {
                    id
                    identifier
                    fullRoundText
                    state
                    slots {
                        entrant {
                            id
                            name
                            seeds {
                                seedNum
                            }
                            participants {
                                id
                                user {
                                    id
                                    slug
                                    name
                                    genderPronoun
                                    location {
                                        city
                                        country
                                        state
                                    }
                                    images(type: "profile") {
                                        url
                                    }
                                }
                                player {
                                    id
                                    gamerTag
                                    prefix
                                }
                            }
                        }
                    }
                    phaseGroup {
                        displayIdentifier
                        phase {
                            name
                            groupCount
                        }
                    }
                }
                stream {
                    streamName
                    streamSource
                }
            }
        }
    }
}
